Gradient Descent: Multi-Layer Perceptron Example¶

Suppose we have 2-dimensional data, where there are two classes; and these two classes are divided into two clusters. All four clusters have Gaussian distributions.

This data is known as eXclusive OR (XOR) data; i.e. the data belongs to the positive class iff (if and only if) one input is positive and the other input is negative.

Positive Class:

  • $ Mean(Quadrant_2) = \left(\begin{array}{cc} -1 & 1 \end{array}\right) $
  • $ Mean(Quadrant_4) = \left(\begin{array}{cc} 1 & -1 \end{array}\right) $

Negative Class:

  • $ Mean(Quadrant_1) = \left(\begin{array}{cc} 1 & 1 \end{array}\right) $
  • $ Mean(Quadrant_3) = \left(\begin{array}{cc} -1 & -1 \end{array}\right) $

$ Covariance(Quadrant_1) = Covariance(Quadrant_2) = Covariance(Quadrant_3) = Covariance(Quadrant_4) = \left(\begin{array}{cc} 0.1 & 0.0\\ 0.0 & 0.1 \end{array}\right) $

# Create the synthetic data
import numpy as np

count = 1000
proportion_quadrant = 0.25

X_q1 = np.random.multivariate_normal(mean = [   1,   1 ],
                                     cov = [ [ 0.1, 0.0 ], [ 0.0, 0.1 ] ],
                                     size = int(proportion_quadrant * count))
X_q2 = np.random.multivariate_normal(mean = [  -1,   1 ],
                                     cov = [ [ 0.1, 0.0 ], [ 0.0, 0.1 ] ],
                                     size = int(proportion_quadrant * count))
X_q3 = np.random.multivariate_normal(mean = [  -1,  -1 ],
                                     cov = [ [ 0.1, 0.0 ], [ 0.0, 0.1 ] ],
                                     size = int(proportion_quadrant * count))
X_q4 = np.random.multivariate_normal(mean = [   1,  -1 ],
                                     cov = [ [ 0.1, 0.0 ], [ 0.0, 0.1 ] ],
                                     size = int(proportion_quadrant * count))

X = np.concatenate((X_q2, X_q4, X_q1, X_q3), axis = 0).astype("float32")
Y = np.array([1]*int(2 * proportion_quadrant * count) + [0]*int(2 * proportion_quadrant * count)).astype("float32")
index = np.arange(count)
np.random.shuffle(index)

trnCount = int(0.8 * count)
valCount = int(0.1 * count)
tstCount = int(0.1 * count)

trnX = X[index[:trnCount]]
trnY = Y[index[:trnCount]]
valX = X[index[trnCount:(trnCount+valCount)]]
valY = Y[index[trnCount:(trnCount+valCount)]]
tstX = X[index[(trnCount+valCount):(trnCount+valCount+tstCount)]]
tstY = Y[index[(trnCount+valCount):(trnCount+valCount+tstCount)]]

print("trn shapes:", trnX.shape, trnY.shape, sep = "\t")
print("val shapes:", valX.shape, valY.shape, sep = "\t")
print("tst shapes:", tstX.shape, tstY.shape, sep = "\t")

import tensorflow
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense
from tensorflow.keras.optimizers import SGD
tensorflow.random.set_seed(1)

model = Sequential()
model.add(Dense(2, activation = "relu", input_shape=(2,)))
model.add(Dense(1, activation = "sigmoid"))
model.summary()

model.compile(loss = "binary_crossentropy", optimizer = SGD(), metrics = [ "accuracy" ])
old_weights = model.get_weights()

print("\ncomparing predictions:")
prediction = model.predict(trnX[:1])[0,0]
print(np.round(prediction, 6))

np0 = np.float32(0)
np1 = np.float32(1)
def sigmoid(x):
    return np1 / (np1 + np.exp(-x))

activation_layer_1 = np.maximum(np0, np.dot(trnX[:1], old_weights[0]) + old_weights[1])
activation_layer_2 = sigmoid(np.dot(activation_layer_1, old_weights[2]) + old_weights[3])[0,0]
print(np.round(activation_layer_2, 6))

model.fit(trnX[:1], trnY[:1], validation_data = (valX, valY), epochs = 1, batch_size = 1)
new_weights = model.get_weights()

print("\nold weights and biases:")
print(old_weights[0])
print(old_weights[1])
print(old_weights[2])
print(old_weights[3])

print("\nnew weights and biases:")
